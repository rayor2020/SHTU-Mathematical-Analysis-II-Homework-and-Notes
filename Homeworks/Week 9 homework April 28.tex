(2) $\displaystyle\iiint_V xy^2 z^3 \, \mathrm{d}x\mathrm{d}y\mathrm{d}z$, $V$: 由 $z=xy$, $y=x$, $x=1$, $z=0$ 围成;
\begin{sol}
    该区域位于第一卦限, 积分区域可表示为 $0\leq x\leq 1$, $0\leq y\leq x$, $0\leq z\leq xy$.

    因此

    \begin{align*}
        \iiint_V xy^2 z^3 \, \mathrm{d}x\mathrm{d}y\mathrm{d}z
         & =\int_0^1\,\mathrm{d}x\int_0^x\,\mathrm{d}y\int_0^{xy}xy^2z^3\,\mathrm{d}z \\
         & =\frac{1}{4}\int_0^1\,\mathrm{d}x\int_0^xx^5y^6\,\mathrm{d}y               \\
         & =\frac{1}{28}\int_0^1x^{12}\,\mathrm{d}x                                   \\
         & =\frac{1}{364}
    \end{align*}
\end{sol}
(4) $\displaystyle\iiint_V (a-y) \, \mathrm{d}x\mathrm{d}y\mathrm{d}z$, $V$: 由 $y=0$, $z=0$, $2x+y=a$, $x+y=a$, $y+z=a$ 围成.
\begin{sol}
    积分区域满足 $0\leq y\leq a$, $\dfrac{a-y}{2}\leq x\leq a-y$, $0\leq z\leq a-y$.

    因此

    \begin{align*}
        \iiint_V (a-y) \, \mathrm{d}x\mathrm{d}y\mathrm{d}z
         & =\int_0^a\,\mathrm{d}y\int_{\frac{a-y}{2}}^{a-y}\,\mathrm{d}x\int_0^{a-y}(a-y)\,\mathrm{d}z \\
         & =\int_0^a\,\mathrm{d}y\int_{\frac{a-y}{2}}^{a-y}(a-y)^2\,\mathrm{d}x                        \\
         & =\frac{1}{2}\int_0^a(a-y)^3\,\mathrm{d}y                                                    \\
         & =\frac{a^4}{8}
    \end{align*}
\end{sol}

(2) $\displaystyle\int_{-R}^{R}\,\mathrm{d}x \int_{-\sqrt{R^2-x^2}}^{\sqrt{R^2-x^2}}\,\mathrm{d}y \int_{0}^{\sqrt{R^2-x^2-y^2}} (x^2+y^2) \, \mathrm{d}z$;
\begin{sol}
    采用柱面坐标变换: 记 $\begin{cases}
            x=r\cos\theta \\
            y=r\sin\theta \\
            z=z
        \end{cases}$, 则积分区域为 $\begin{cases}
            0\leq r\leq R        \\
            0\leq\theta\leq 2\pi \\
            0\leq z\leq \sqrt{R^2-r^2}
        \end{cases}$, 雅可比行列式为 $\dfrac{\partial(x,y,z)}{\partial(r,\theta,z)}=r$.

    因此

    \begin{align*}
        \int_{-R}^{R}\,\mathrm{d}x \int_{-\sqrt{R^2-x^2}}^{\sqrt{R^2-x^2}}\,\mathrm{d}y \int_{0}^{\sqrt{R^2-x^2-y^2}} (x^2+y^2) \, \mathrm{d}z
         & =\int_0^R\,\mathrm{d}r\int_0^{2\pi}\,\mathrm{d}\theta\int_0^{\sqrt{R^2-r^2}}r^3\,\mathrm{d}z \\
         & =\int_0^R\,\mathrm{d}r\int_0^{2\pi}r^3\sqrt{R^2-r^2}\,\mathrm{d}\theta                       \\
         & =2\pi\int_0^R r^3\sqrt{R^2-r^2}\,\mathrm{d}r
    \end{align*}

    对 $\displaystyle\int r^3\sqrt{R^2-r^2}\,\mathrm{d}r$ 采用分部积分法. 记 $u=r^2$, $\mathrm{d}v=r\sqrt{R^2-r^2}\,\mathrm{d}r$, 则 $\mathrm{d}u=2r\,\mathrm{d}r$, $v=-\dfrac{1}{3}(R^2-r^2)^{\frac{3}{2}}$.

    因此

    \begin{align*}
        \int r^3\sqrt{R^2-r^2}\,\mathrm{d}r
         & =\int u\,\mathrm{d}v                                                                         \\
         & =uv-\int v\,\mathrm{d}u                                                                      \\
         & =-\frac{r^2}{3}(R^2-r^2)^{\frac{3}{2}}+\frac{2}{3}\int r(R^2-r^2)^{\frac{3}{2}}\,\mathrm{d}r \\
         & =-\frac{r^2}{3}(R^2-r^2)^{\frac{3}{2}}-\frac{2}{15}(R^2-r^2)^{\frac{5}{2}}+C
    \end{align*}

    从而

    \begin{align*}
        \int_0^R r^3\sqrt{R^2-r^2}\,\mathrm{d}r
         & =\left.\left(-\frac{r^2}{3}(R^2-r^2)^{\frac{3}{2}}-\frac{2}{15}(R^2-r^2)^{\frac{5}{2}}\right)\right|_0^R \\
         & =\frac{2}{15}R^5
    \end{align*}

    因此

    \begin{align*}
        \text{原式}
         & =2\pi\int_0^R r^3\sqrt{R^2-r^2}\,\mathrm{d}r \\
         & =\frac{4}{15}\pi R^5
    \end{align*}
\end{sol}

7. 设$\displaystyle F(t) = \iiint\limits_{x^2+y^2+z^2 \leq t^2} f(x^2+y^2+z^2) \, \mathrm{d}x\mathrm{d}y\mathrm{d}z$, 其中 $f$ 为可微分函数, 求 $F'(t)$.

    因此

    \begin{align*}
        \iiint\limits_{x^2+y^2+z^2 \leq t^2} f(x^2+y^2+z^2) \, \mathrm{d}x\mathrm{d}y\mathrm{d}z
         & =\int_0^t f(r^2)r^2\,\mathrm{d}r\int_0^{\pi}\sin\theta\,\mathrm{d}\theta\int_0^{2\pi}\,\mathrm{d}\varphi \\
         & =4\pi\int_0^t f(r^2)r^2\,\mathrm{d}r
    \end{align*}

    由变上限积分求导法则可得 $F'(t)=4\pi t^2 f(t^2)$.
\end{sol}


8. 证明: $\displaystyle\iiint\limits_{x^2+y^2+z^2 \leq 1} f(z) \, \mathrm{d}v = \pi \int_{-1}^{1} f(z)(1-z^2) \, \mathrm{d}z$.
\begin{pf}
    考虑等式左边的三重积分, 先对 $x$, $y$ 积分.

    对于固定的 $z$, 截面区域为 $x^2+y^2\leq 1-z^2$. 采用柱面坐标变换: 记 $\begin{cases}
            x=r\cos\theta \\
            y=r\sin\theta \\
        \end{cases}$, 则 $\begin{cases}
            0\leq r\leq\sqrt{1-z^2} \\
            0\leq\theta\leq 2\pi
        \end{cases}$, 雅可比行列式为 $\dfrac{\partial(x,y)}{\partial(r,\theta)}=r$.

    因此

    \begin{align*}
        \iiint\limits_{x^2+y^2+z^2 \leq 1} f(z) \, \mathrm{d}x\mathrm{d}y\mathrm{d}z
         & =\int_{-1}^1 f(z)\,\mathrm{d}z\iint\limits_{x^2+y^2\leq 1-z^2}\,\mathrm{d}x\mathrm{d}y           \\
         & =\int_{-1}^1 f(z)\,\mathrm{d}z\int_0^{\sqrt{1-z^2}}r\,\mathrm{d}r\int_0^{2\pi}\,\mathrm{d}\theta \\
         & =\pi\int_{-1}^1 f(z)(1-z^2)\,\mathrm{d}z
    \end{align*}

    这与等式右边相等.

    Q.E.D.
\end{pf}


12. 一个物体是由两个半径各为 $R$ 和 $r$ $(R>r)$ 的同心球所围成, 已知材料的密度和到球心的距离成反比, 且在距离等于 $1$ 处的密度等于 $k$, 求物体的总质量.
\begin{sol}

            x=\rho\sin\theta\cos\varphi \\
            y=\rho\sin\theta\sin\varphi \\
            z=\rho\cos\theta
        \end{cases}$, 则体积元为 $\rho^2\sin\theta\,\mathrm{d}\rho\mathrm{d}\theta\mathrm{d}\varphi$.

    由题意可知, 在点 $(\rho,\theta,\varphi)$ 处的密度为 $\dfrac{k}{\rho}$.

    因此物体的总质量为

    $$2\pi\int_r^R\frac{k}{\rho} \cdot \rho^2 \, \mathrm{d}\rho\int_0^{\pi}\sin\theta\,\mathrm{d}\theta=2k\pi(R^2-r^2)$$
\end{sol}

13. 半径为 $a$ 的圆盘, 其各点的密度和到圆心的距离成正比 (比例系数为 $1$), 今内切于圆盘截去半径为 $\frac{a}{2}$ 之小圆, 求余下部分的重心坐标.
\begin{sol}
    由对称性可知, 重心位于两圆圆心连成的直线上. 以该直线为 $x$ 轴, 大圆圆心为坐标原点, 垂直方向为 $y$ 轴建立坐标系, 则只需计算重心在 $x$ 轴方向上的坐标.

    坐标 $(x,y)$ 处的密度为 $\sqrt{x^2+y^2}$.

    需要计算剩余部分的力矩
    $$\iint\limits_{x^2+y^2\leq a^2}x\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y-\iint\limits_{\left(x-\frac{a}{2}\right)^2+y^2\leq\frac{a^2}{4}}x\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y$$
    和质量
    $$\iint\limits_{x^2+y^2\leq a^2}\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y-\iint\limits_{\left(x-\frac{a}{2}\right)^2+y^2\leq\frac{a^2}{4}}\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y$$

    采用柱面坐标变换: 记 $\begin{cases}
            x=r\cos\theta \\
            y=r\sin\theta
        \end{cases}$, 则雅可比行列式为 $\dfrac{\partial(x,y)}{\partial(r,\theta)}=r$.

    对于大圆区域, 积分区域为 $\begin{cases}
            0\leq r\leq a \\
            0\leq \theta\leq 2\pi
        \end{cases}$.

    对于小圆区域, 由 $\left(x-\dfrac{a}{2}\right)^2+y^2\leq\dfrac{a^2}{4}$ 可得 $r^2-ar\cos\theta+\dfrac{a^2}{4}\leq\dfrac{a^2}{4}$, 即 $r\leq a\cos\theta$. 因此积分区域为 $\begin{cases}
            -\frac{\pi}{2}\leq \theta\leq \frac{\pi}{2} \\
            0\leq r\leq a\cos \theta
        \end{cases}$.

    首先计算大圆的力矩和质量:

    \begin{align*}
        \text{力矩}
         & =\iint\limits_{x^2+y^2\leq a^2}x\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y \\
         & =\int_0^a r^3\,\mathrm{d}r\int_0^{2\pi}\cos\theta\,\mathrm{d}\theta    \\
         & =\frac{a^4}{4}\cdot0                                                   \\
         & =0
    \end{align*}

    这是符合直觉的.

    \begin{align*}
        \text{质量}
         & =\iint\limits_{x^2+y^2\leq a^2}\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y \\
         & =\int_0^a r^2\,\mathrm{d}r\int_0^{2\pi}\,\mathrm{d}\theta             \\
         & =\frac{2a^3\pi}{3}
    \end{align*}

    接下来计算小圆的力矩:

    \begin{align*}
        \text{力矩}
         & =\iint\limits_{\left(x-\frac{a}{2}\right)^2+y^2\leq\frac{a^2}{4}}x\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y \\
         & =\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\,\mathrm{d}\theta\int_0^{a\cos\theta}r^3\cos\theta\,\mathrm{d}r   \\
         & =\frac{a^4}{4}\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\cos^5\theta\,\mathrm{d}\theta
    \end{align*}

    计算 $\displaystyle\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\cos^5\theta\,\mathrm{d}\theta$: 由于被积函数为偶函数, 因此

    \begin{align*}
        \int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\cos^5\theta\,\mathrm{d}\theta
         & =2\int_0^{\frac{\pi}{2}}\cos^5\theta\,\mathrm{d}\theta          \\
         & =2\int_0^1(1-u^2)^2\,\mathrm{d}u                                \\
         & =2\int_0^1 (1-2u^2+u^4)\,\mathrm{d}u                            \\
         & =2\left.\left(u-\frac{2}{3}u^3+\frac{1}{5}u^5\right)\right|_0^1 \\
         & =\frac{16}{15}
    \end{align*}

    其中采用了换元法: 记 $u=\sin\theta$, 则 $\mathrm{d}u=\cos\theta\,\mathrm{d}\theta$.

    因此小圆的力矩为 $\dfrac{4a^4}{15}$.

    再计算小圆的质量:

    \begin{align*}
        \text{质量}
         & =\iint\limits_{\left(x-\frac{a}{2}\right)^2+y^2\leq\frac{a^2}{4}}\sqrt{x^2+y^2}\,\mathrm{d}x\mathrm{d}y \\
         & =\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\,\mathrm{d}\theta\int_0^{a\cos\theta}r^2\,\mathrm{d}r            \\
         & =\frac{a^3}{3}\int_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\cos^3\theta\,\mathrm{d}\theta                       \\
         & =\frac{4a^3}{9}
    \end{align*}

    因此剩余部分的重心 $x$ 坐标为

    $$\frac{0-\frac{4a^4}{15}}{\frac{2a^3\pi}{3}-\frac{4a^3}{9}}=\frac{6a}{10-15\pi}$$

    即重心坐标为 $\left(\dfrac{6a}{10-15\pi},0\right)$.
\end{sol}
