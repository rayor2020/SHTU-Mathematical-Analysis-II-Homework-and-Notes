\begin{theorem}
    Let $D$ and $D'$ be two bounded open set on $\mathbb{R}^2$, whose boundaries consist of piecewisely $C^1$ curves. Suppose $x=x(u,v)$, $y=y(u,v)$ is a $C^1$ map from $D'$ to $D$ which is one-to-one and $f(x,y)$ is continuous on $D$, moreover assume $\frac{\partial(x,y)}{\partial(u,v)}\neq 0$ on $D'$, then

    $$\iint_Df(x,y)\,\mathrm{d}x\mathrm{d}y=\iint_{D'}f\left(x(u,v),y(u,v)\right)\left|\frac{\partial(x,y)}{\partial(u,v)}\right|\,\mathrm{d}u\mathrm{d}v$$
\end{theorem}

$$
    \raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/02/0101}}\quad

        \overset{r:\,\begin{cases}\substack{x=x(u,v)\\y=y(u,v)}\end{cases}}{\underset{\substack{1:1 \\C^1\text{ map}\\\boxed{\scriptstyle\frac{\partial (x,y)}{\partial(u,v)}\neq 0}}}{\longrightarrow}}
        \quad\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/10/02/0102}}\quad
        \overset{f}{\longrightarrow}                                                                \\
        \overset{\tilde{f}=f\circ r}{\longrightarrow}

$$

Take limit about the partition $T'$ (rectangular), then we obtain
$$
    \iint_Df(x,y)\,\mathrm{d}x\mathrm{d}y=\iint_{D'}f\left(x(u,v),y(u,v)\right)\left|\frac{\partial(x,y)}{\partial(u,v)}\right|\,\mathrm{d}u\mathrm{d}v
$$

        \text{Vol}
         & =2\iint_{\frac{x^2}{a^2}+\frac{y^2}{b^2}\leq 1}c\sqrt{1-\frac{x^2}{a^2}-\frac{y^2}{b^2}}\,\mathrm{d}x\mathrm{d}y                                   \\

         & =2abc\int_0^1\,\mathrm{d}r\int_0^{2\pi}r\sqrt{1-r^2}\,\mathrm{d}\theta                                                                             \\

    \end{align*}

\begin{example}
    Find the area enclosed by the leminiscate $(x^2+y^2)^2=2a^2(x^2-y^2)$

    \begin{center}
        \raisebox{-0.5\height}{\includegraphics[height=2.5cm]{chapters/10/02/0107}}\quad$\begin{cases}
                x=r\cos\theta \\
                y=r\sin\theta
            \end{cases}$
    \end{center}

    $r^4=2a^2r^2\cos2\theta$

    $r^2=2a^2\cos2\theta$

    \begin{align*}
        \text{Area}
         & =\iint_D1\,\mathrm{d}x\mathrm{d}y
        =2\int_{-\frac{\pi}{4}}^{\frac{\pi}{4}}\frac{1}{2}r^2(\theta)\,\mathrm{d}\theta                    \\
         & =2\iint_{\substack{-\frac{\pi}{4}\leq\theta\leq\frac{\pi}{4}                                    \\0\leq r\leq\sqrt{2a^2\cos2\theta}}}r\,\mathrm{d}r\mathrm{d}\theta\\
         & =\int_{-\frac{\pi}{4}}^{\frac{\pi}{4}}\,\mathrm{d}\theta\int_0^{2a^2\cos2\theta}2r\,\mathrm{d}r \\
         & =\int_{-\frac{\pi}{4}}^{\frac{\pi}{4}}2a^2\cos2\theta\,\mathrm{d}\theta                         \\
         & =2a^2
    \end{align*}

    $$\begin{cases}
            \frac{\partial f}{\partial x}=2(x^2+y^2)\cdot2x-2a^2\cdot 2x=4x(x^2+y^2-a^2) \\
            \frac{\partial f}{\partial y}=4y(x^2+y^2+a^2)
        \end{cases}$$

    $\implies (0,0)$ is the only possible singularity of the curve.
\end{example}
