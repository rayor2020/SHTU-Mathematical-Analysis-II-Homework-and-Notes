\subsection[三重积分]{Triple Integral}

$$\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/10/03/0101}}\quad\overset{f}{\rightarrow}\mathbb{R}$$

For any \emph{partition} $T=\left\{\Omega_i\middle|i\in\Lambda\right\}$ of $\Omega$, if $\lim_{\|T\|\to 0}\sum_{i\in\Lambda}f(M_i)\sigma(\Omega_i)$ exists, then its limit is called the triple integral of $f$ on $\Omega$, and denote it by $\iiint_{\Omega}f\,\underset{\substack{\downarrow\\\text{``volume element'' 体积元}}}{\mathrm{d}V}=\iiint_{\Omega}f\,\mathrm{d}x\mathrm{d}y\mathrm{d}z=\iiint_{\Omega}f(x,y,z)\,\mathrm{d}x\mathrm{d}y\mathrm{d}z$

\subparagraph*{Fact}

If $f$ is integrable, then $f$ is ``bounded''.

\subparagraph*{Recall}

$\displaystyle \iint_{\Omega}f(x,y)\,\mathrm{d}x\mathrm{d}y=\lim\sum f(M_i)\sigma(\Omega_i)$

If $f$ represents the \emph{``surface density''}, then $\iint_{\Omega}f\,\mathrm{d}\sigma$ represents the total \emph{mass}.

If $f:\,\Omega\to\mathbb{R}$ represents the (volume) density, then $\iiint_{\Omega}f\,\mathrm{d}V$ represents the total mass.

\subsubsection[三重积分的累次积分]{Computation}

\begin{enumerate}
    \item The Simplest region $\Omega=[a_1,b_1]\times[a_2,b_2]\times[a_3,b_3]=\left\{(x,y,z)\in\mathbb{R}^3\middle|\substack{a_1\leq x\leq b_1,\\a_2\leq y\leq b_2,\\a_3\leq z\leq b_3}\right\}$, $f$ is integrable on $\Omega$, then

          \begin{align*}
              \iiint_{[a_1,b_1]\times[a_2,b_2]\times[a_3,b_3]}f(x,y,z)\,\mathrm{d}x\mathrm{d}y\mathrm{d}z
               & \overset{3=2+1}{=}\int_{a_1}^{b_1}\,\mathrm{d}x\iint_{[a_2,b_2]\times[a_3,b_3]}f(x,y,z)\,\mathrm{d}y\mathrm{d}z\quad\text{``竖切''} \\
               & =\int_{a_2}^{b_2}\,\mathrm{d}y\iint_{[a_1,b_1]\times[a_3,b_3]}f(x,y,z)\,\mathrm{d}x\mathrm{d}z\quad\text{``侧切''}                  \\
               & =\int_{a_3}^{b_3}\,\mathrm{d}z\iint_{[a_1,b_1]\times[a_2,b_2]}f(x,y,z)\,\mathrm{d}x\mathrm{d}y\quad\text{``横切''}                  \\
               & \overset{3=1+2}{=}\iint_{[a_1,b_1]\times[a_2,b_2]}\,\mathrm{d}x\mathrm{d}y\int_{a_3}^{b_3}f(x,y,z)\,\mathrm{d}z                   \\
               & =\iint_{[a_2,b_2]\times[a_3,b_3]}\,\mathrm{d}y\mathrm{d}z\int_{a_1}^{b_1}f(x,y,z)\,\mathrm{d}x                                    \\
               & =\iint_{[a_1,b_1]\times[a_3,b_3]}\,\mathrm{d}x\mathrm{d}z\int_{a_2}^{b_2}f(x,y,z)\,\mathrm{d}y
          \end{align*}

    \item More general region

          \raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/03/0102}}\quad$\displaystyle\implies\int_a^b\,\mathrm{d}x\int_{y_1(x)}^{y_2(x)}f(x,y)\,\mathrm{d}y$

          \raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/03/0103}}\quad\allbold{$\displaystyle I_x=\left\{y\in\mathbb{R}\middle|(x,y)\in\Omega\right\}$}

          $$
              \iint_{\Omega}f(x,y)\,\mathrm{d}x\mathrm{d}y=\int_a^b\,\mathrm{d}x\int_{I_x}f(x,y)\,\mathrm{d}y
          $$

          (where $a=\inf_{\exists(x,y)\in\Omega} x$, $b=\sup_{\exists(x,y)\in\Omega} x$)
\end{enumerate}

\subparagraph*{3-dimensional}

\begin{enumerate}

\end{enumerate}

\subsubsection[三重积分的换元]{Substitution}

$$\raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/03/0106}}
    \quad\overset{\substack{\text{one-to-one}\\C^1}}{\underset{\substack{\boldsymbol{\overrightarrow{r}(u,v,w)=}\left\{\substack{x=x(u,v,w) \\
                    y=y(u,v,w) \\
                    z=z(u,v,w)}\right.\\\frac{\partial(x,y,z)}{\partial(u,v,w)}\neq 0}}{\longrightarrow}}\quad
    \raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/03/0107}}
    \quad\overset{f}{\longrightarrow}\mathbb{R}$$

\allbold{微元法 $\Delta V\approx\left|\left<\overrightarrow{r_u}\Delta u\times\overrightarrow{r_v}\Delta v,\overrightarrow{r_w}\Delta w\right>\right|=\left|\frac{\partial(x,y,z)}{\partial(u,v,w)}\right|\Delta u\Delta v\Delta w$}

$$
    \iiint_{\Omega}f(x,y,z)\,\mathrm{d}x\mathrm{d}y\mathrm{d}z=\iiint_{\Omega'}f\left(x(u,v,w),y(u,v,w),z(u,v,w)\right)\left|\frac{\partial(x,y,z)}{\partial(u,v,w)}\right|\,\mathrm{d}u\mathrm{d}v\mathrm{d}w
$$

\allbold{$\displaystyle \mathrm{d}V=\mathrm{d}x\mathrm{d}y\mathrm{d}z=\left|\frac{\partial(x,y,z)}{\partial(u,v,w)}\right|\,\mathrm{d}u\mathrm{d}v\mathrm{d}w$}

\begin{example}
    Let $V$ be the tetrahedron 四面体 enclosed by $x=0$, $y=0$, $z=0$ and $x+y+z=1$. Find the integral $\displaystyle\iiint_V\frac{\mathrm{d}x\mathrm{d}y\mathrm{d}z}{(1+x+y+z)^3}=\int_0^1\,\mathrm{d}z\iint_{\substack{x\geq0,y\geq0\\x+y\leq1-z}}\frac{\mathrm{d}x\mathrm{d}y}{(1+x+y+z)^3}$

    \begin{center}
        \includegraphics[height=3cm]{chapters/10/03/0108}
    \end{center}

    \begin{enumerate}
        \item  \begin{align*}
                  \iiint_V\frac{\mathrm{d}x\mathrm{d}y\mathrm{d}z}{(1+x+y+z)^3}
                   & =\iint_{\substack{x\geq 0,y\geq 0                                                                                  \\x+y\leq 1}}\,\mathrm{d}x\mathrm{d}y\int_{0\leq z\leq1-x-y}\frac{\mathrm{d}z}{(1+x+y+z)^3}\\
                   & =\iint_{\substack{x\geq 0,y\geq 0                                                                                  \\x+y\leq 1}}\left(-\frac{1}{2}\cdot\left.\frac{1}{(1+x+y+z)^2}\right|_0^{1-x-y}\right)\,\mathrm{d}x\mathrm{d}y\\
                   & =-\frac{1}{2}\iint_{x\geq 0,y\geq 0,x+y\leq 1}\left(\frac{1}{4}-\frac{1}{(1+x+y)^2}\right)\,\mathrm{d}x\mathrm{d}y \\
                   & =-\frac{1}{2}\int_0^1\,\mathrm{d}x\int_0^{1-x}\left(\frac{1}{4}-\frac{1}{(1+x+y)^2}\right)\,\mathrm{d}y            \\
                   & =-\frac{1}{2}\int_0^1\left.\left(\frac{1}{4}y+\frac{1}{1+x+y}\right)\right|_0^{1-x}\,\mathrm{d}x                   \\
                   & =-\frac{1}{2}\int_0^1\left(\frac{1-x}{4}+\frac{1}{2}-\frac{1}{1+x}\right)\,\mathrm{d}x                             \\
                   & =-\frac{1}{2}\left(\frac{3}{4}-\frac{1}{8}-\ln 2\right)
              \end{align*}

        \item $$\begin{cases}
                      u=x+y+z \\
                      v=y     \\
                      w=z
                  \end{cases}\quad\begin{cases}
                      v\geq 0       \\
                      w\geq 0       \\
                      0\leq u\leq 1 \\
                      u-(v+w)\geq 0
                  \end{cases}$$

              $\displaystyle\frac{\partial(u,v,w)}{\partial(x,y,z)}=1\implies\frac{\partial(x,y,z)}{\partial(u,v,w)}=1$\quad\raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/10/03/0109}}

              \begin{align*}
                  I
                   & =\iiint_{\substack{0\leq u\leq 1                                                                        \\v,w\geq 0\\u\geq v+w}}\frac{1}{(1+u)^3}\,\mathrm{d}u\mathrm{d}v\mathrm{d}w\\
                   & =\int_0^1\frac{\mathrm{d}u}{(1+u)^3}\iint_{\substack{v,w\geq 0                                          \\v+w\leq u}}\,\mathrm{d}v\mathrm{d}w\\
                   & =\int_0^1\frac{u^2}{2(1+u)^3}\,\mathrm{d}u                                                              \\
                   & =\frac{1}{2}\int_0^1\frac{u^2+u-(u+1)+1}{(1+u)^3}\,\mathrm{d}u                                          \\
                   & =\frac{1}{2}\int_0^1\left(\frac{u+1-1}{(1+u)^2}-\frac{1}{(1+u)^2}+\frac{1}{(1+u)^3}\right)\,\mathrm{d}u \\
                   & =\frac{1}{2}\int_0^1\left(\frac{1}{1+u}-\frac{2}{(1+u)^2}+\frac{1}{(1+u)^3}\right)\,\mathrm{d}u         \\
                   & =\frac{1}{2}(\ln 2-1+\frac{3}{8})
              \end{align*}
    \end{enumerate}
\end{example}

    \end{align*}

                y=b\cdot r\cdot\sin\theta
            \end{cases}$, $\begin{cases}

Substitution $\begin{cases}
        x=a\cdot r\cdot\sin\theta\cdot\cos\varphi \\
        y=b\cdot r\cdot\sin\theta\cdot\sin\varphi \\
        z=c\cdot r\cdot\cos\theta
    \end{cases}$, $\begin{cases}
        r\in[0,1]        \\
        \theta\in[0,\pi] \\
        \varphi\in[0,2\pi]
    \end{cases}$\quad\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/10/03/0110}}\quad$(a,b,c)=(1,1,1)$

Jacobi matrix $\underset{\boldsymbol{=\begin{pmatrix}
                \frac{\partial\overrightarrow{F}}{\partial r} & \frac{\partial\overrightarrow{F}}{\partial\theta} & \frac{\partial\overrightarrow{F}}{\partial\varphi}
            \end{pmatrix}}}{\begin{pmatrix}
            a\sin\theta\cos\varphi & a r\cos\theta\cos\varphi & -a r\sin\theta\sin\varphi \\
            b\sin\theta\sin\varphi & b r\cos\theta\sin\varphi & b r\sin\theta\cos\varphi  \\
            c\cos\theta            & -c r\sin\theta           & 0
        \end{pmatrix}}\implies\frac{\partial(x,y,z)}{\partial(r,\theta,\varphi)}=abc\cdot r^2\sin\theta$

\raisebox{-0.5\height}{\includegraphics[height=4cm]{chapters/10/03/0111}}\quad$\begin{cases}
        \left|\frac{\partial\overrightarrow{F}}{\partial r}\right|=1                \\
        \left|\frac{\partial\overrightarrow{F}}{\partial\varphi}\right|=r\sin\theta \\
        \left|\frac{\partial\overrightarrow{F}}{\partial\theta}\right|=r
    \end{cases}$

$\implies$ Integral$\displaystyle=\iiint_{\substack{0\leq r\leq 1\\0\leq\theta\leq\pi\\0\leq\varphi\leq 2\pi}}(a^2\sin^2\theta\cos^2\varphi+b^2\sin^2\theta\sin^2\varphi+c^2\cos^2\theta)\cdot r^2\cdot r^2\sin\theta\,\mathrm{d}r\mathrm{d}\theta\mathrm{d}\varphi$

For standard spherical coordinate, $\mathrm{d}V=\mathrm{d}x\mathrm{d}y\mathrm{d}z=r^2\sin\theta\,\mathrm{d}r\mathrm{d}\theta\mathrm{d}\varphi$
