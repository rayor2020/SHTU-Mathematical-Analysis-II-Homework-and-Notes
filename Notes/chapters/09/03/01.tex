\begin{thm}[Implicit Function Theorem]
    Let $D\subseteq\mathbb{R}^2$ be an open set, $M_0=(x_0,y_0)\in D$. Let $F$ be defined on $D$ and satisfy\begin{enumerate}
        \item $F\in C^1(D)$ \allbold{连续可微 ($\frac{\partial F}{\partial x},\frac{\partial F}{\partial y}$ are continuous)}
        \item $F(x_0,y_0)=0$
        \item $\frac{\partial F}{\partial y}(x_0,y_0)\neq 0$
    \end{enumerate}
    $\implies$ rectangular neighborhood $I\times J$ of $(x_0,y_0)$ s.t. $F(x,y)=0$ determines \allbold{an \emph{implicit function} $y=y(x)$ in $I\times J$}. In other words, $\forall x\in I$, $\exists!y\in J$, s.t. $F(x,y)=0$. Moreover, \allbold{$y=y(x)$ is continuously differentiable} and \allbold{$\left.\frac{dy}{dx}\right|_x=-\left.\frac{\frac{\partial F}{\partial x}(x,y)}{\frac{\partial F}{\partial y}(x,y)}\right|_{y=y(x)}$}.
\end{thm}

\begin{example}
    Use Implicit Function Theorem to understand $x^2+y^2-1=0$

    $F(x,y)\triangleq x^2+y^2-1$

    $\begin{cases}
            \frac{\partial F}{\partial y}(x,y)=2y=0 \\
            F(x,y)=x^2+y^2-1=0
        \end{cases}\implies(x,y)=(1,0)$, or $(-1,0)$

    By the IFT, for any $(x_0,y_0)\in C\setminus\{(1,0),(-1,0)\}$, $\exists$ locally defined implicit function $y=y(x)$, and $y'(x)=-\left.\frac{2x}{2y}\right|_{y=y(x)}=-\left.\frac{x}{y}\right|_{y=y(x)}$

    \begin{enumerate}
        \item $\left(\frac{1}{2},\frac{\sqrt{3}}{2}\right)$, $y'(x)=-\frac{\frac{1}{2}}{\frac{\sqrt{3}}{2}}=-\frac{1}{\sqrt{3}}\rightarrow$ tangent line $y-\frac{\sqrt{3}}{2}=-\frac{1}{\sqrt{3}}\left(x-\frac{1}{2}\right)$
        \item $\left(\frac{1}{2},-\frac{\sqrt{3}}{2}\right)$, $y'(x)=-\frac{\frac{1}{2}}{-\frac{\sqrt{3}}{2}}=\frac{1}{\sqrt{3}}$
    \end{enumerate}
    Explicitly, $y=\sqrt{1-x^2}$, $y'=-\frac{x}{\sqrt{1-x^2}}$

    $y''(x)=\frac{d}{dx}(-\frac{x}{y})=-\frac{y-x\frac{dy}{dx}}{y^2}=-\frac{y+x\cdot\frac{x}{y}}{y^2}=-\frac{x^2+y^2}{y^3}=-\frac{1}{y^3}$
\end{example}

    $$
        \begin{cases}
            \frac{\partial F}{\partial y}(x,y)=3y^2-3x=0 \\
            F(x,y)=x^3+y^3-3xy=0
        \end{cases}\rightarrow\substack{y^6+y^3-3y^3=0\\y^3(y^3-2)=0}\implies(x,y)=\begin{cases}
            (0,0) \\
            \left(\sqrt[3]{4},\sqrt[3]{2}\right)
        \end{cases}
    $$

    Around $(x_0,y_0)\in C\setminus\left\{(0,0),\left(\sqrt[3]{4},\sqrt[3]{2}\right)\right\}$, we can apply Implicit Function Thm to obtain $y=y(x)$, and $\left.\frac{dy}{dx}\right|_x=-\left.\frac{3x^2-3y}{3y^2-3x}\right|_{y=y(x)}=-\left.\frac{x^2-y}{y^2-x}\right|_{y=y(x)}$

    Try to find the critical point of $y=y(x)$
    $$
        \left.\frac{dy}{dx}\right|_x=0\iff\begin{cases}
            -\frac{x^2-y}{y^2-x}=0 \\
            F(x,y)=0
        \end{cases}\implies\begin{cases}
            x^2=y \\
            x^3+y^3-3xy=0
        \end{cases}\rightarrow(x,y)=\begin{cases}
            (0,0) \\
            \left(\sqrt[3]{2},\sqrt[3]{4}\right)
        \end{cases}
    $$

    It follows that the critical point for $y=y(x)$ is $x=\sqrt[3]{2}$

    Now\begin{align*}
        \left.\frac{d^2y}{dx^2}\right|_{x=\sqrt[3]{2}} & =\left.\frac{d}{dx}\left(-\frac{x^2-y}{y^2-x}\right)\right|_{x=\sqrt[3]{2}}                                                      \\
                                                       & =-\left.\frac{\left(2x-\frac{dy}{dx}\right)(y^2-x)-(x^2-y)\left(2y\cdot\frac{dy}{dx}-1\right)}{(y^2-x)^2}\right|_{x=\sqrt[3]{2}} \\
                                                       & =-\frac{2\sqrt[3]{2}\left(\sqrt[3]{16}-\sqrt[3]{2}\right)}{\left(\sqrt[3]{16}-\sqrt[3]{2}\right)^2}<0.
    \end{align*}
    $\leadsto x=\sqrt[3]{2}$ is a local maximum. $\left(\sqrt[3]{2},\sqrt[3]{4}\right)$\quad\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/09/03/0101}}
\end{example}

\subparagraph*{Method 1}

Suppose $\frac{\partial F}{\partial y}(x_0,y_0)>0$\quad\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/09/03/0103}}

\begin{pf}
    Step 1. Suppose $\frac{\partial F}{\partial y}(x_0,y_0)=\lim_{y\to y_0}\frac{F(x_0,y)-F(x_0,y_0)}{y-y_0}>0$, we have $\frac{F(x_0,y)-F(x_0,y_0)}{y-y_0}>0$, $\forall y\in[y_0-b,y_0+b]$ where $b>0$ is sufficiently small. Then $\begin{cases}
            F(x_0,y)>0 & y\in(y_0,y_0+b] \\
            F(x_0,y)<0 & y\in[y_0-b,y_0)
        \end{cases}$

    Step 2. If we take $a>0$ sufficiently small, then $\begin{cases}
            F(x,y_0-b)<0 \\
            F(x,y_0+b)>0
        \end{cases}$, $\forall x\in[x_0-a,x_0+a]$

    \allbold{(Where we use the continuity of the functions $F(\cdot,y_0+b),F(\cdot,y_0-b)$)}

    Then $\forall x\in[x_0-a,x_0+a]$, $\exists y\in[y_0-b,y_0+b]$ s.t. $F(x,y)=0$ (By the Intermediate Value Theorem for the function $F(x,\cdot)$ on the interval $[y_0-b,y_0+b]$)

    Since $F\in C^1$, $\frac{\partial F}{\partial y}$ is continuous, so $\frac{\partial F}{\partial y}>0$ on $B(M_0,r)$ for $r>0$ small enough.

    Then we can take $b>0$ small enough, and then $a>0$ small enough to guarentee $[x_0-a,x_0+a]\times[y_0-b,y_0+b]\subseteq B(M_0,r)$

    \begin{center}
        $(r,b,a)$\quad\raisebox{-0.5\height}{\includegraphics[height=3cm]{chapters/09/03/0104}}
    \end{center}

    Consequently, $\forall x\in[x_0-a,x_0+a]$, $\exists!y\in[y_0-b,y_0+b]$ s.t. $F(x,y)=0$.
    \begin{rmk}
        \emph{The arguments go through if $x$ is replaced by a vector in $\mathbb{R}^n$ while keeping $y$ in $\mathbb{R}$}
    \end{rmk}

    Step 3. If $\{x_n\}\subseteq[x_0-a,x_0+a]$, $x_n\to x'\,(n\to+\infty)$, then $y(x_n)\to y(x')\,(n\to+\infty)$.

    (Since $\{y(x_n)\}$ has a convergent subsequence, $y(x_{n_k})\to y'$ by B.W.. Moreover $F(x',y')=\lim_{k\to+\infty}F(x_{n_k},y(x_{n_k}))=0\implies y'=y(x')$ by the uniqueness in last step $\implies y(x_n)\to y(x')$ (Continuity of the implicit function $y=y(x)$))

    $\boxed{\text{Fact: If any subsequence of $\{z_n\}$ has a subsequence convergent to $z_0$, then $z_n\to z_0$.}}$

    Step 4. $\begin{cases}
            F(x_0+\Delta x,y_0+\Delta y)=0 \\
            F(x_0,y_0)=0
        \end{cases}$\quad\raisebox{-0.5\height}{\includegraphics[height=2cm]{chapters/09/03/0105}}
    \begin{align*}
        \implies 0 & =F(x_0+\Delta x,y_0+\Delta y)-F(x_0,y_0)                                                                                                                                                                             \\
                   & =F(x_0+\Delta x,y_0+\Delta y)-F(x_0,y_0+\Delta y)+F(x_0,y_0+\Delta y)-F(x_0,y_0)                                                                                                                                     \\
                   & \overset{\text{Here we just use the existence of partial derivatives and LMVT}}{=\frac{\partial F}{\partial x}(X_0+\theta\Delta x,y_0+\Delta y)\Delta x+\frac{\partial F}{\partial y}(x_0,y_0+\eta\Delta y)\Delta y}
    \end{align*}
    \begin{align*}
        \implies\Delta y & =-\frac{\frac{\partial F}{\partial x}(x_0+\theta\Delta x,y_0+\Delta y)}{\frac{\partial F}{\partial y}(x_0,y_0+\eta\Delta y)}\Delta x \\
                         & \overset{\substack{\text{continuity of }\frac{\partial F}{\partial x},\frac{\partial F}{\partial y}                                  \\\downarrow}}{=}-\frac{\frac{\partial F}{\partial x}(x_0,y_0)}{\frac{\partial F}{\partial y}(x_0,y_0)}\Delta x+\circ(\Delta x)\text{, }\Delta x\to0
    \end{align*}

    $\implies y=y(x)$ is differentiable at $x_0$ and $\left.\frac{dy}{dx}\right|_{x=x_0}=-\left.\frac{\frac{\partial F}{\partial x}}{\frac{\partial F}{\partial y}}\right|_{(x_0,y_0)}$

    \begin{align*}
        0 & =F(x_0+\Delta x,y_0+\Delta y)-F(x_0,y_0)                                                                                                                             \\
          & =\frac{\partial F}{\partial x}(x_0,y_0)\Delta x+\frac{\partial F}{\partial y}(x_0,y_0)\Delta y+\circ(\sqrt{(\Delta x)^2+(\Delta y)^2})\text{, }\Delta x,\Delta y\to0
    \end{align*}

    \emph{$\left.\frac{dy}{dx}\right|_x=-\left.\frac{\frac{\partial F}{\partial x}}{\frac{\partial F}{\partial y}}\right|_{y=y(x)}$ In particular, $y\in C^1[x_0-a,x_0+a]$}
\end{pf}

\begin{rmk}
    If you have already $y=y(x)$ exists, and differentiable, then we can differentiate
    $$F(x,y(x))=0\quad(\substack{x\mapsto(x,y(x))\\(x,y)\mapsto F(x,y)})$$

    $\frac{\partial F}{\partial x}dx+\frac{\partial F}{\partial y}dy=0\implies\frac{dy}{dx}=-\left.\frac{\frac{\partial F}{\partial x}}{\frac{\partial F}{\partial y}}\right|_{y=y(x)}$
\end{rmk}
